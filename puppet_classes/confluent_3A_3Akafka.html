  • package_name (Any) (defaults to: 'confluent-kafka-2.11') —

class confluent::kafka(
  $package_ensure='installed',
  $package_name='confluent-kafka-2.11'
)  {

  package{ $package_name:
    alias  => 'kafka',
    ensure => $package_ensure
  } -> Ini_setting <| tag == 'kafka-setting' |>
}
